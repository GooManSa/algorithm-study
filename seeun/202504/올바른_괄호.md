function solution(s){
    const stack = [];
    
    for(let ele of s){
        if(ele === '('){
            stack.push(ele);
        }else if (ele === ')'){
            if(stack.length === 0) return false;
            stack.pop();
        }
    }
    return stack.length === 0;
}
